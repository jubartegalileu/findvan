## Technical Details

### Database Schema

```sql
CREATE TABLE leads (
  id BIGSERIAL PRIMARY KEY,
  source TEXT NOT NULL,
  name TEXT NOT NULL,
  phone VARCHAR(15),
  email VARCHAR(255),
  address TEXT,
  city TEXT NOT NULL,
  company_name TEXT,
  cnpj VARCHAR(20),
  url TEXT,
  captured_at TIMESTAMP DEFAULT NOW(),
  is_valid BOOLEAN DEFAULT true,
  is_duplicate BOOLEAN DEFAULT false,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW(),

  UNIQUE(phone, source),
  CONSTRAINT valid_phone_length CHECK (phone IS NULL OR LENGTH(phone) = 11)
);

CREATE INDEX idx_leads_phone ON leads(phone);
CREATE INDEX idx_leads_city ON leads(city);
CREATE INDEX idx_leads_source ON leads(source);
CREATE INDEX idx_leads_valid ON leads(is_valid);
CREATE INDEX idx_leads_created ON leads(created_at DESC);
```

### API Functions

```javascript
// Insert leads from scraper
insertLeads(leads, source)
  → { inserted: 10, duplicate: 2, errors: 0 }

// Query functions
getLeadsByCity(city, page=1, limit=10)
  → { leads: [], total: 45, page: 1, pages: 5 }

getValidLeads(city=null, page=1)
  → { leads: [], total: 40, is_valid: true }

getDuplicates()
  → { phone: "11987654321", count: 3, records: [] }

// Deduplication
mergeDuplicates(phone, keepId)
  → { merged: 2, deleted: 1 }
```
